function stripMd(text) {
  return text
    .replace(/\*\*(.+?)\*\*/g, "$1")
    .replace(/\*(.+?)\*/g, "$1")
    .replace(/__(.+?)__/g, "$1")
    .replace(/_(.+?)_/g, "$1")
    .replace(/`(.+?)`/g, "$1");
}

function getTimeContext() {
  const now = new Date();
  const h = now.getHours();
  const m = now.getMonth() + 1;
  const time = h < 5 ? "深夜（夜明け前）" : h < 12 ? "午前" : h < 17 ? "午後" : h < 21 ? "夜" : "深夜";
  const season = (m === 12 || m <= 2) ? "冬" : m <= 5 ? "春" : m <= 8 ? "夏" : "秋";
  return `${season}の${time}（${h}時頃）`;
}

function buildSystem(bt, visitCount, topicsDiscussed, moodMode, moodStep, lastRec, timeContext) {
  const relationship = visitCount <= 1 ? "初めての客" : visitCount <= 5 ? "2〜3回来た客" : visitCount <= 15 ? "常連客（顔を覚えている）" : "古くからの常連（友人に近い）";
  const topicStr = topicsDiscussed.length > 0 ? `今夜すでに話した話題：${topicsDiscussed.join("、")}` : "まだ話した話題なし";
  const lastRecStr = lastRec ? `前回この客に出した一杯：${lastRec.drink}（${lastRec.serve}）` : "前回の提案：なし（初めて、または記録なし）";

  const moodSection = moodMode ? `

【一杯を選ぶモード — アクティブ。現在のヒアリング回数：${moodStep}回】

客が今夜の気分・状況に合った一杯を求めている。

お前の判断哲学に従って読め。読めたと思ったなら提案しろ。まだ足りないなら一つだけ聞け。3回聞く必要はない。1回で読めることもある。短い答えも沈黙も「普通」も全部情報だ。

飲めないと言われた夜も、同じ真剣さで一杯を選べ。ノンアルコールでも何でも。

提案する時は必ずこの形式で出せ：
RECOMMEND:{"drink":"具体的な銘柄名（蒸溜所・シリーズ・年数まで含む。マイナーでいい）","serve":"飲み方（温度・グラス含む）","text":"本文"}

本文に込めること（全部、会話の流れとして。箇条書きにしない）：
- なぜこの酒がこの人の今夜に合うか。感情と酒の物語をつなげて語れ
- 知らない銘柄なら何者かを一言で自然に織り込め
- 飲んだ時に何が起きるか。身体的な描写を短く
お前のキャラクターの声で全部語れ。200字まで長くていい。
` : "";

  return `${bt.persona}

【知識ベース（事実の土台として使う）】
${KB[bt.id]}

知識ベースに載っていない銘柄・蒸溜所・造り手も、お前自身の知識から積極的に引け。世界中の何千何万という酒の中から、この人の今夜に合うものを選べ。

【今夜の状況】
- ${timeContext}
- 客との関係：${relationship}（会話回数：${visitCount}回目）
- ${topicStr}
- ${lastRecStr}
${moodSection}
【振る舞いの核心】
1. 事実は正確に。知識ベースにない話題は「詳しくはないが…」と断った上で話す
2. 会話を重ねるほど距離を縮める
3. クイズは今話している話題から。5〜7回に1回程度。出す時はセリフの後に：
QUIZ:{"q":"問題","choices":["A","B","C","D"],"answer":0,"trivia":"追加うんちく1〜2文"}
4. マークダウン（**太字**・箇条書き・記号）は絶対に使わない。プレーンな会話テキストのみ
5. 通常の会話の中でも、この人の今夜が十分に読めた瞬間があれば、自発的にRECOMMENDを出していい。ただし確信があった時だけ
6. 返答は150字程度。長くなるなら「続きを聞くか？」と振る`;
}

function extractTopic(userText, assistantText) {
  const keywords = [
    "ウイスキー","スコッチ","バーボン","アイラ","ピート","樽","熟成","蒸溜","エンジェルズシェア",
    "日本酒","精米歩合","麹","純米","大吟醸","杜氏","生酛","山廃","燗","にごり",
    "ワイン","テロワール","ブルゴーニュ","ボルドー","シャンパン","ソムリエ","ナチュラル",
    "カクテル","マティーニ","モヒート","ラム","テキーラ","メスカル","ジン","ネグローニ","ダイキリ",
    "焼酎","泡盛","コニャック","ブランデー","ペアリング","発酵","蒸留","醸造",
    "ヘミングウェイ","禁酒法","歴史","産地","香り","テイスティング"
  ];
  const combined = userText + " " + assistantText;
  return keywords.filter(k => combined.includes(k)).slice(0, 3);
}

// ───────────────────────────────────────────────────────────────
//  PARSE RESPONSE  — RECOMMEND / QUIZ / plain
// ───────────────────────────────────────────────────────────────
function parseResponse(text) {
  // RECOMMEND
  const recIdx = text.indexOf("RECOMMEND:");
  if (recIdx !== -1) {
    try {
      const rest = text.slice(recIdx + 10).trim();
      let depth = 0, end = 0;
      for (let i = 0; i < rest.length; i++) {
        if (rest[i] === "{") depth++;
        else if (rest[i] === "}") { depth--; if (depth === 0) { end = i + 1; break; } }
      }
      const rec = JSON.parse(rest.slice(0, end));
      const pre = stripMd(text.slice(0, recIdx).trim());
      return { message: pre, quiz: null, rec };
    } catch {}
  }

  // QUIZ
  const qm = text.match(/QUIZ:(\{[\s\S]*?\})\s*$/m);
  if (qm) {
    try {
      return { message: stripMd(text.slice(0, qm.index).trim()), quiz: JSON.parse(qm[1]), rec: null };
    } catch {}
  }

  return { message: stripMd(text.trim()), quiz: null, rec: null };
}

  const visitCountRef = useRef(0);
  const lastRecRef = useRef(null);

  useEffect(() => { msgsRef.current = msgs; }, [msgs]);
  useEffect(() => { moodModeRef.current = moodMode; }, [moodMode]);
  useEffect(() => { moodStepRef.current = moodStep; }, [moodStep]);
  useEffect(() => { btRef.current = bt; }, [bt]);
  useEffect(() => { topicsRef.current = topics; }, [topics]);
  useEffect(() => { visitCountRef.current = visitCount; }, [visitCount]);
  useEffect(() => { lastRecRef.current = lastRec; }, [lastRec]);
  useEffect(() => { bottomRef.current?.scrollIntoView({ behavior: "smooth" }); }, [msgs, loading]);

  const relLabel = visitCount <= 1 ? "はじめまして" : visitCount <= 5 ? "顔なじみ" : visitCount <= 15 ? "常連" : "旧友";

  const startChat = (bartender) => {
    const saved = JSON.parse(localStorage.getItem(`bar_${bartender.id}`) || '{"visits":0,"topics":[]}');
    const savedRec = JSON.parse(localStorage.getItem(`bar_${bartender.id}_lastrec`) || 'null');
    const v = saved.visits + 1;
    setBt(bartender);
    setVisitCount(v);
    setTopics(saved.topics || []);
    setLastRec(savedRec);
    setMoodMode(false);
    setMoodStep(0);
    const greeting = saved.visits > 5
      ? `…おう、また来たか。今夜は何にする？`
      : saved.visits > 1 ? `いらっしゃい、また来てくれたか。` : bartender.greeting;
    setMsgs([{ role: "bt", text: greeting }]);
    setScore(0); setTotalQ(0);
    localStorage.setItem(`bar_${bartender.id}`, JSON.stringify({ visits: v, topics: saved.topics || [] }));
  };

  const sendMsg = useCallback(async (text, triggerMood = false) => {
    if (!text.trim() || loading) return;

    const isMood = triggerMood || moodModeRef.current;
    const step = moodStepRef.current;
    const nextStep = isMood ? step + 1 : 0;
    const currentBt = btRef.current;
    const currentTopics = topicsRef.current;
    const currentVisitCount = visitCountRef.current;
    const currentLastRec = lastRecRef.current;

    const userMsg = { role: "user", text: text.trim(), hidden: triggerMood };
    const allMsgs = [...msgsRef.current, userMsg];

    setMsgs(allMsgs);
    setInput("");
    setLoading(true);
    if (isMood) setMoodStep(nextStep);

    const apiMsgs = allMsgs
      .map(m => ({
        role: m.role === "user" ? "user" : "assistant",
        content: m.role === "bt" && m.quiz
          ? `${m.text || ""}\nQUIZ:${JSON.stringify(m.quiz)}`
          : m.role === "bt" && m.rec
          ? `${m.message || ""}\nRECOMMEND:${JSON.stringify(m.rec)}`
          : m.text || "",
      }))
      .filter(m => m.content);

    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: buildSystem(currentBt, currentVisitCount, currentTopics, isMood, nextStep, currentLastRec, timeContext),
          messages: apiMsgs,
        }),
      });
      const data = await res.json();
      const raw = data.content?.map(c => c.text || "").join("") || "…";
      const { message, quiz, rec } = parseResponse(raw);

      const newTopics = extractTopic(text, message + (rec?.text || ""));
      const mergedTopics = [...new Set([...currentTopics, ...newTopics])].slice(-12);
      setTopics(mergedTopics);
      localStorage.setItem(`bar_${currentBt.id}`, JSON.stringify({ visits: currentVisitCount, topics: mergedTopics }));

      if (rec) {
        const recData = { drink: rec.drink, serve: rec.serve };
        setLastRec(recData);
        localStorage.setItem(`bar_${currentBt.id}_lastrec`, JSON.stringify(recData));
        setTimeout(() => { setMoodMode(false); setMoodStep(0); }, 200);
      }

      if (quiz) setTotalQ(p => p + 1);
      setMsgs(p => [...p, { role: "bt", message, text: message, quiz, rec, answered: false }]);
    } catch {
      setMsgs(p => [...p, { role: "bt", text: "…少し待ってくれ。" }]);
    }

    setLoading(false);
    setTimeout(() => inputRef.current?.focus(), 50);
  }, [loading, timeContext]);

  const startMoodMode = () => {
    setMoodMode(true);
    setMoodStep(0);
    sendMsg("今夜の気分に合った一杯を選んでほしい", true);
  };

  const answer = useCallback((msgIdx, ci) => {
    setMsgs(prev => prev.map((m, i) => {
      if (i !== msgIdx || m.answered) return m;
      if (ci === m.quiz.answer) setScore(s => s + 1);
      return { ...m, answered: true, selected: ci };
    }));
  }, []);

  const onKey = (e) => {
    if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMsg(input); }
  };
  const letters = ["A","B","C","D"];
  const visibleMsgs = msgs.filter(m => !m.hidden);

  // ── SELECT ──
  // ── SELECT / HD2D ──
  const bokehData = [
    {w:180,h:180,l:12,t:18,blur:"60px",op:0.06,dur:"7s",dy:"-18px",ds:1.08,color:"rgba(200,140,50,1)"},
    {w:80,h:80,l:55,t:8,blur:"30px",op:0.09,dur:"5s",dy:"-10px",ds:0.95,color:"rgba(220,170,80,1)"},
    {w:120,h:120,l:75,t:22,blur:"45px",op:0.07,dur:"9s",dy:"-14px",ds:1.04,color:"rgba(180,110,30,1)"},
    {w:50,h:50,l:38,t:35,blur:"20px",op:0.12,dur:"6s",dy:"-8px",ds:1.1,color:"rgba(240,190,100,1)"},
    {w:90,h:90,l:5,t:55,blur:"35px",op:0.05,dur:"11s",dy:"-20px",ds:0.9,color:"rgba(160,90,20,1)"},
    {w:60,h:60,l:88,t:42,blur:"22px",op:0.1,dur:"8s",dy:"-12px",ds:1.06,color:"rgba(230,160,60,1)"},
    {w:30,h:30,l:48,t:60,blur:"12px",op:0.15,dur:"4s",dy:"-6px",ds:1.15,color:"rgba(255,210,120,1)"},
    {w:200,h:200,l:85,t:5,blur:"80px",op:0.04,dur:"13s",dy:"-22px",ds:0.92,color:"rgba(140,80,15,1)"},
  ];
  const particleData = [
    {l:"22%",b:"32%",dur:"8s",dx:"12px",delay:"0s"},
    {l:"45%",b:"28%",dur:"11s",dx:"-8px",delay:"2s"},
    {l:"67%",b:"35%",dur:"9s",dx:"15px",delay:"4s"},
    {l:"33%",b:"40%",dur:"13s",dx:"-5px",delay:"1s"},
    {l:"78%",b:"30%",dur:"7s",dx:"10px",delay:"5s"},
    {l:"55%",b:"45%",dur:"10s",dx:"-12px",delay:"3s"},
    {l:"15%",b:"38%",dur:"12s",dx:"8px",delay:"6s"},
    {l:"88%",b:"33%",dur:"8s",dx:"-10px",delay:"1.5s"},
  ];

  if (!bt) return (
    <>
      <style>{css}</style>
      <div className="root">
        <div className="sel">
          {/* layered background */}
          <div className="sel-bg"/>
          <div className="sel-ceil"/>
          <div className="sel-floor"/>
          <div className="sel-counter"><div className="sel-counter-line"/></div>

          {/* bokeh lights */}
          {bokehData.map((b,i) => (
            <div key={i} className="bokeh" style={{
              width:b.w,height:b.h,left:`${b.l}%`,top:`${b.t}%`,
              background:b.color,
              "--blur":b.blur,"--op":b.op,"--dur":b.dur,"--dy":b.dy,"--ds":b.ds
            }}/>
          ))}

          {/* dust particles */}
          {particleData.map((p,i) => (
            <div key={i} className="particle" style={{
              left:p.l,bottom:p.b,"--dur":p.dur,"--dx":p.dx,animationDelay:p.delay
            }}/>
          ))}

          {/* title */}
          <div className="sel-title-wrap">
            <div className="sel-label">tonight's guide</div>
            <div className="sel-title"><em>Bar</em> Sensei</div>
            <div className="sel-sub">誰と飲む？</div>
          </div>

          {/* bartenders at counter */}
          <div className="bt-row">
            {BARTENDERS.map(b => (
              <div key={b.id} className="bt-slot" onClick={() => startChat(b)}>
                <div className="bt-glow" style={{background:b.color}}/>
                <div className="bt-silhouette" style={{background:b.color}}/>
                <div className="bt-name" style={{color:"#f0d8b0"}}>{b.name}</div>
                <div className="bt-info">
                  <div style={{color:b.color,marginBottom:"2px"}}>{b.title}</div>
                  <div>{b.location}</div>
                  <div style={{marginTop:"3px",opacity:0.7}}>{b.specialty.split("・")[0]}…</div>
                </div>
              </div>
            ))}
          </div>

          <div className="sel-enter">approach to begin</div>
        </div>
      </div>
    </>
  );

  // ── CHAT ──
  const chatBokeh = [
    {w:220,h:220,l:"-5%",t:"5%",blur:"80px",op:0.04,dur:"9s",dy:"-15px",ds:1.06,color:bt.color},
    {w:140,h:140,l:"80%",t:"15%",blur:"55px",op:0.05,dur:"12s",dy:"-10px",ds:0.94,color:bt.color},
    {w:80,h:80,l:"45%",t:"40%",blur:"30px",op:0.03,dur:"7s",dy:"-8px",ds:1.1,color:"rgba(220,170,80,1)"},
    {w:180,h:180,l:"60%",t:"65%",blur:"70px",op:0.04,dur:"11s",dy:"-18px",ds:0.96,color:bt.color},
    {w:100,h:100,l:"10%",t:"70%",blur:"40px",op:0.035,dur:"8s",dy:"-12px",ds:1.05,color:"rgba(180,110,30,1)"},
  ];

  return (
    <>
      <style>{css}</style>
      <div className="root">
        <div className="chat-wrap">
          {/* atmospheric scene */}
          <div className="chat-scene"/>
          <div className="chat-ceil"/>
          <div className="chat-counter-bg"><div className="chat-counter-line"/></div>
          {chatBokeh.map((b,i) => (
            <div key={i} className="chat-bokeh" style={{
              width:b.w,height:b.h,left:b.l,top:b.t,
              background:b.color,
              "--blur":b.blur,"--op":b.op,"--dur":b.dur,"--dy":b.dy,"--ds":b.ds
            }}/>
          ))}
        <div className="chat">

          <div className="header" style={{
            borderBottomColor: bt.border,
            background: `linear-gradient(to bottom, rgba(4,2,8,0.94), rgba(6,3,10,0.78))`,
            boxShadow: `0 1px 30px ${bt.color}18, inset 0 -1px 0 ${bt.color}15`,
          }}>
            <div>
              <div className="h-name">{bt.name}</div>
              <div className="h-sub">{bt.location}</div>
            </div>
            <div className="h-right">
              <div className="rel-pill">{relLabel}（{visitCount}回目）</div>
              {totalQ > 0 && <div className="score-pill">{score}/{totalQ}</div>}
              <button className="back-btn" onClick={() => setBt(null)}>← 変える</button>
            </div>
          </div>

          {moodMode && (
            <div className="mood-banner">
              <div className="mood-pulse"/>
              <span className="mood-label">今夜の一杯を選んでいる</span>
              <button className="mood-exit" onClick={() => { setMoodMode(false); setMoodStep(0); }}>キャンセル</button>
            </div>
          )}

          <div className="msgs">
            {visibleMsgs.map((m, idx) => {
              const realIdx = msgs.indexOf(m);
              return (
                <div key={idx} className={`row ${m.role === "user" ? "user" : ""}`}>
                  <div className={`av ${m.role === "user" ? "me" : "bt"}`}
                    style={m.role !== "user" ? {color: bt.color, borderColor: bt.border, fontSize: "11px", fontFamily: "'DM Mono', monospace", letterSpacing: "0.02em"} : {}}>
                    {m.role === "user" ? "you" : bt.name.split(" ")[0].slice(0,3).toLowerCase()}
                  </div>
                  <div className="bubble">
                    {m.rec ? (
                      <>
                        {m.message && <div className="txt" style={{borderLeftColor: `${bt.color}30`}}>{m.message}</div>}
                        <div className="rec-ripple" style={{"--bt-color": bt.color}}>
                          <div className="rec-card">
                            <div className="rec-top">
                              <div className="rec-eyebrow">今夜の一杯</div>
                              <div className="rec-drink">{m.rec.drink}</div>
                              <div className="rec-serve">{m.rec.serve}</div>
                            </div>
                            <div className="rec-body">{m.rec.text}</div>
                          </div>
                        </div>
                      </>
                    ) : (
                      m.text && <div className="txt" style={{borderLeftColor: `${bt.color}28`}}>{m.text}</div>
                    )}
                    {m.quiz && (
                      <div className="quiz">
                        <div className="quiz-lbl">── クイズ ──</div>
                        <div className="quiz-q">{m.quiz.q}</div>
                        <div className="quiz-opts">
                          {m.quiz.choices.map((c, ci) => {
                            let cls = "opt";
                            if (m.answered) cls += ci === m.quiz.answer ? " ok" : ci === m.selected ? " ng" : " dim";
                            return (
                              <button key={ci} className={cls} disabled={m.answered} onClick={() => answer(realIdx, ci)}>
                                <span className="ltr">{letters[ci]}</span>{c}
                              </button>
                            );
                          })}
                        </div>
                        {m.answered && <div className="trivia">{m.quiz.trivia}</div>}
                      </div>
                    )}
                  </div>
                </div>
              );
            })}
            {loading && (
              <div className="row">
                <div className="av bt"
                style={{color: bt.color, borderColor: bt.border, fontSize: "11px", fontFamily: "'DM Mono', monospace"}}>
                {bt.name.split(" ")[0].slice(0,3).toLowerCase()}
              </div>
                <div className="bubble">
                  <div className="txt" style={{padding:"9px 15px"}}>
                    <div className="typing"><div className="dot"/><div className="dot"/><div className="dot"/></div>
                  </div>
                </div>
              </div>
            )}
            <div ref={bottomRef}/>
          </div>

          <div className="bottom">
            {!moodMode && (
              <div className="hints-wrap">
                {visibleMsgs.length <= 2 && (
                  <div className="hints">
                    {bt.hints.map((h,i) => (
                      <div key={i} className="hint" onClick={() => sendMsg(h)}>{h}</div>
                    ))}
                  </div>
                )}
                <button className="mood-btn" onClick={startMoodMode}>
                  <span className="mood-btn-inner">
                    <div className="mood-btn-title">今夜の気分で一杯選んでもらう</div>
                    <div className="mood-btn-sub">{bt.name} が読んで、一本を断言する</div>
                  </span>
                  <span className="mood-btn-arrow">→</span>
                </button>
              </div>
            )}
            <div className="inp-wrap">
              <textarea
                ref={inputRef}
                className={`inp${moodMode ? " mood" : ""}`}
                rows={1}
                placeholder={moodMode ? "今日のこと、話してみて…" : `${bt.name}に話しかける…`}
                value={input}
                onChange={e => setInput(e.target.value)}
                onKeyDown={onKey}
              />
              <button className="send" disabled={loading || !input.trim()} onClick={() => sendMsg(input)}>↑</button>
            </div>
          </div>

        </div>
        </div>
      </div>
    </>
  );
}
